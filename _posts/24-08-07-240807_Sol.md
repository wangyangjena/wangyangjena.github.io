## A



题解略, 提供一种找约数的方法



首先, 一个数的约数是关于 $\sqrt{x}$ 对称的, 即若 $a * b = x$ 则 $a \le \sqrt{x} \le b \le x$



所以就有了一个简单的 $O(\sqrt{n})$ 的找约数方法



```cpp
    int i = 0;
    for(i = 1; i * i < k; i++){
        if(k % i == 0){

            st.push_back(k / i);

        st.push_back(i);
```

### 代码



```cpp
#include<bits/stdc++.h>
#define int long long
#define Pii pair <long long, long long>
using namespace std;
const int N = 1e5 + 10, MOD = 998244353;

int n, m, k;
int a[N], b[N], suma[N], sumb[N];

vector <int> rec;
vector <int> st;
int tmpa[N], tmpb[N], ca[N], cb[N];

void Calc1(){
    int i = 0;
    for(i = 1; i * i < k; i++){
        if(k % i == 0){
            st.push_back(i);
            st.push_back(k / i);
        }
    }
    if(i * i == k){
        st.push_back(i);
    }
    sort(st.begin(), st.end());
    
    for(int i = 1; i <= n; i++){
        suma[i] = suma[i - 1] + a[i];
    }

    for(int i = 1; i <= m; i++){
        sumb[i] = sumb[i - 1] + b[i];
    }

    ++tmpa[0];
    for(int i = 1; i <= n; i++){
        for(auto j : st){
            if(suma[i] < j) break;
            ca[j] += tmpa[suma[i] - j];
        }
        ++tmpa[suma[i]];
    }

    ++tmpb[0];
    for(int i = 1; i <= m; i++){
        for(auto j : st){
            if(sumb[i] < j) break;
            cb[j] += tmpb[sumb[i] - j];
        }
        ++tmpb[sumb[i]];
    }    

    int ans = 0;
    for(auto i : st){
        int j = k / i;
        if(j > m || i > n) continue;
        ans += (ca[i] * cb[j]) % MOD;
        ans = (ans > MOD) ? ans - MOD : ans;
    }
    cout << ans << "\n";
}

signed main(){
    cin >> n >> m >> k;
    for(int i = 1; i <= n; i++){
        cin >> a[i];
    }
    for(int i = 1; i <= m; i++){
        cin >> b[i];
    }
    Calc1();
    // printf("%.10lf\n", 1.0 * clock() / CLOCKS_PER_SEC);
}
```





## B



首先发现是一棵基环树, 然后直接贪心即可



用 Topo Sort 找环会容易一点, 



### 代码

```cpp
#include<bits/stdc++.h>
#define int long long
#define Pii pair <long long, long long>
using namespace std;
const int N = 1e6 + 10, INF = 1e18, Upt = 1e10;

int id, t;
int n, m, l;
int u, v, d;

Pii operator + (Pii a, Pii b){
    Pii ret = {0, 0};
    ret.first = a.first + b.first;
    ret.second = a.second + b.second;
    return ret;
}

struct Segment_Tree{
    struct Node{
        int ls, rs;
        int sum, cnt;
    }t[N * 4];
    int rt = 1, cnt = 1;

    void Modify(int &id, int l, int r, int ps){
        if(!id) id = ++cnt;
        t[id].sum += ps;
        ++t[id].cnt;
        
        if(l == r) return;
        int mid = (l + r) / 2;
        if(ps <= mid) Modify(t[id].ls, l, mid, ps);
        else Modify(t[id].rs, mid + 1, r, ps);
    }

    Pii Query(int id, int l, int r, int ql, int qr){
        if(!id) return {0, 0};
        if(ql <= l && r <= qr) return {t[id].sum, t[id].cnt};

        Pii ret = {0, 0};
        int mid = (l + r) / 2;
        if(ql <= mid) ret = ret + Query(t[id].ls, l, mid, ql, qr);
        if(qr > mid) ret = ret + Query(t[id].rs, mid + 1, r, ql, qr);
        return ret;
    }

    void Reset(){
        for(int i = 1; i <= cnt; i++){
            t[i].ls = t[i].rs = 0;
            t[i].sum = t[i].cnt = 0;
        }
        rt = 1, cnt = 1;
    }
}tr;

struct Vec{
    int v, nxt, id;
}e[N];
int head[N], cnte, ide;
map <int, int> recd[N];

struct Edge{
    int u, v, d;

    bool operator < (const Edge &p)const{
        return d < p.d;
    }
}rec[N];
int cnt;

int dfn[N], low[N], cdfn;
bool ins[N];
vector <int> st;
vector <Edge> rdfn;

void Insert(int u, int v){
    e[++cnte].v = v;
    e[cnte].id = ide;
    e[cnte].nxt = head[u];
    head[u] = cnte;
}

void Tarjan(int x, int lst){
    dfn[x] = low[x] = ++cdfn;
    ins[x] = 1, st.push_back(x);
    for(int ii = head[x]; ii; ii = e[ii].nxt){
        int i = e[ii].v;
        if(!dfn[i]){
            Tarjan(i, e[ii].id);
            low[x] = min(low[x], low[i]);
        }else if(ins[i] && e[ii].id != lst){
            low[x] = min(low[x], dfn[i]);
        }
    }

    if(dfn[x] == low[x]){
        int tp = st.back();
        ins[tp] = 0, st.pop_back();

        int lst = tp, tpa = tp;
        while(tp != x){
            tp = st.back(), st.pop_back();
            ins[tp] = 0;
            rdfn.push_back({lst, tp, recd[lst][tp]});
            lst = tp;
        }
        if(tpa != x){
            rdfn.push_back({tpa, x, recd[tpa][x]});
        }
    }
}

bool Check(int x){
    Pii ret = tr.Query(1, 1, Upt, 1, x);
    int add = ret.second * x - ret.first;
    return add <= l;
}

void Reset(){
    cnt = 0, cnte = 0, ide = 0, cdfn = 0;
    rdfn.clear();
    st.clear();
    for(int i = 1; i <= n; i++){
        head[i] = 0;
        ins[i] = 0;
        dfn[i] = low[i] = 0;
        recd[i].clear();
    }
    tr.Reset();
}

void Calc(){
    cin >> n >> m >> l;
    int tag = 0, ru = 0, rv = 0, rd = 0;
    for(int i = 1; i <= m; i++){
        cin >> u >> v >> d;

        if(recd[u][v]){
            tag = 1;
            ru = u, rv = v, rd = d;
            rec[++cnt] = {u, v, -1};
            continue;
        }

        ++ide;
        Insert(u, v);
        Insert(v, u);
        recd[u][v] = recd[v][u] = d;
        rec[++cnt] = {u, v, d};
    }

    if(tag == 1){
        for(int i = 1; i <= m; i++){
            if(((rec[i].u == ru && rec[i].v == rv) || (rec[i].u == rv && rec[i].v == ru)) && rec[i].d != -1){
                rec[i].d += rd;
                break;
            }
        }
    }

    if(!tag && m == n){
        for(int i = 1; i <= n; i++){
            if(!dfn[i]){
                Tarjan(i, 0);
            }
        }
        sort(rdfn.begin(), rdfn.end());

        int du = rdfn[0].u, dv = rdfn[0].v, dd = rdfn[0].d, f = 0;
        int au = rdfn[1].u, av = rdfn[1].v, ff = 0;
        for(int i = 1; i <= m; i++){
            if(!f && ((rec[i].u == du && rec[i].v == dv) || (rec[i].u == dv && rec[i].v == du))){
                rec[i].d = -1;
                f = 1;
                continue;
            }
            if(!ff && ((rec[i].u == au && rec[i].v == av) || (rec[i].u == av && rec[i].v == au))){
                rec[i].d += dd;
                ff = 1;
            }
        }
    }

    for(int i = 1; i <= m; i++){
        if(rec[i].d == -1) continue;
        tr.Modify(tr.rt, 1, Upt, rec[i].d);
    }

    int ll = 1, rr = 1e10;

    while(ll < rr){
        int mid = (ll + rr + 1) / 2;
        if(Check(mid)){
            ll = mid;
        }else{
            rr = mid - 1;
        }
    }
    cout << ll << "\n";

    Reset();
}

signed main(){
    cin >> id >> t;

    while(t--){
        Calc();
    }
    // printf("%.10lf\n", 1.0 * clock() / CLOCKS_PER_SEC);
}
```
